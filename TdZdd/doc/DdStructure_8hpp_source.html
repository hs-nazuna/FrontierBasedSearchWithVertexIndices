308  bool operator==(DdStructure const& o) const {
309  int n = root_.row();
310  if (n != o.root_.row()) return false;
311  if (n == 0) return root_ == o.root_;
312  if (root_ == o.root_ && &*diagram == &*o.diagram) return true;
313  if (size() > o.size()) return o.operator==(*this);

317  {
318  size_t om = (*o.diagram)[0].size();
319  equiv[0].resize(om);
320  for (size_t j = 0; j < om; ++j) {
321  equiv[0][j] = j;
322  }
323  }
324

454  template<typename S, typename T, typename R>
455  R evaluate(DdEval<S,T,R> const& evaluator) const {
456  S eval(evaluator.entity()); // copied
457  bool msg = eval.showMessages();
458  int n = root_.row();
459 
460  MessageHandler mh;
461  if (msg) {
462  mh.begin(typenameof(eval));
463 #ifdef _OPENMP
464  if (useMP) mh << " " << omp_get_max_threads() << "x";
465 #endif
466  mh.setSteps(n);
467  }
468 
469 #ifdef _OPENMP
470  int threads = useMP ? omp_get_max_threads() : 0;
471  MyVector<S> evals(threads, eval);
472 #endif
473  eval.initialize(n);
474 #ifdef _OPENMP
475  if (useMP)
476 #pragma omp parallel
477  {
478  int k = omp_get_thread_num();
479  evals[k].initialize(n);
480  }
481 #endif
482 
483  DataTable<T> work(diagram->numRows());
484  {
485  size_t const m = (*diagram)[0].size();
486  assert(m >= 2);
487  work[0].resize(m);
488  for (size_t j = 0; j < m; ++j) {
489  eval.evalTerminal(work[0][j], j);
490  }
491  }
492 
493  for (int i = 1; i <= n; ++i) {
494  MyVector<Node<ARITY> > const& node = (*diagram)[i];
495  size_t const m = node.size();
496  work[i].resize(m);
497 
498 #ifdef _OPENMP
499  if (useMP)
500 #pragma omp parallel
501  {
502  int k = omp_get_thread_num();
503 
504 #pragma omp for schedule(static)
505  for (size_t j = 0; j < m; ++j) {
506  DdValues<T,ARITY> values;
507  for (int b = 0; b < ARITY; ++b) {
508  NodeId f = node[j].branch[b];
509  values.setReference(b, work[f.row()][f.col()]);
510  values.setLevel(b, f.row());
511  }
512  evals[k].evalNode(work[i][j], i, values);
513  }
514  }
515  else
516 #endif
517  for (size_t j = 0; j < m; ++j) {
518  DdValues<T,ARITY> values;
519  for (int b = 0; b < ARITY; ++b) {
520  NodeId f = node[j].branch[b];
521  values.setReference(b, work[f.row()][f.col()]);
522  values.setLevel(b, f.row());
523  }
524  eval.evalNode(work[i][j], i, values);
525  }
526 
527  MyVector<int> const& levels = diagram->lowerLevels(i);
528  for (int const* t = levels.begin(); t != levels.end(); ++t) {
529  work[*t].clear();
530  eval.destructLevel(*t);
531  }
532 #ifdef _OPENMP
533  if (useMP)
534 #pragma omp parallel
535  {
536  int k = omp_get_thread_num();
537  for (int const* t = levels.begin(); t != levels.end(); ++t) {
538  evals[k].destructLevel(*t);
539  }
540  }
541 #endif
542  if (msg) mh.step();
543  }
544 
545  R retval = eval.getValue(work[root_.row()][root_.col()]);
546  if (msg) mh.end();
547  return retval;
548  }

603  private:
604  void next(NodeId f) {
605  itemset.clear();
606 
607  for (;;) {
608  while (f > 1) { /* down */
609  Node<ARITY> const& s = (*dd.diagram)[f.row()][f.col()];
610 
611  if (s.branch[0] != 0) {
612  cursor = path.size();
613  path.push_back(Selection(f, false));
614  f = s.branch[0];
615  }
616  else {
617  path.push_back(Selection(f, true));
618  f = s.branch[1];
619  }
620  }
621 
622  if (f == 1) break; /* found */
623 
624  for (; cursor >= 0; --cursor) { /* up */
625  Selection& sel = path[cursor];
626  Node<ARITY> const& ss =
627  (*dd.diagram)[sel.node.row()][sel.node.col()];
628  if (sel.val == false && ss.branch[1] != 0) {
629  f = sel.node;
630  sel.val = true;
631  path.resize(cursor + 1);
632  f = dd.diagram->child(f, 1);
633  break;
634  }
635  }
636 
637  if (cursor < 0) { /* end() state */
638  cursor = -2;
639  path.clear();
640  return;
641  }
642  }
643 
644  for (size_t i = 0; i < path.size(); ++i) {
645  if (path[i].val) {
646  itemset.insert(path[i].node.row());
647  }
648  }
649  }
650  };
